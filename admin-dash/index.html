Google Sheets Integration Setup

      
    

    

      
Follow these steps to automatically backup all user, reservation, and subscriber data to your own Google Sheet:

      
    
        
  1. Create a new Google Sheet.
    2. 
        
  2. Go to Extensions > Apps Script.
    3. 
        
  3. Delete any existing code and paste the following Google Apps Script code:
          
    function doGet(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var result = {};
    var sheets = ss.getSheets();
    
    for (var s = 0; s < sheets.length; s++) {
      var sheet = sheets[s];
      var name = sheet.getName();
      
      if (name === "system_config") {
        var lastRow = sheet.getLastRow();
        if (lastRow > 1) {
          var values = sheet.getRange(2, 1, lastRow - 1, 2).getValues();
          for (var i = 0; i < values.length; i++) {
            var key = values[i][0];
            var rawVal = values[i][1];
            try {
              result[key] = JSON.parse(rawVal);
            } catch(err) {
              result[key] = rawVal;
            }
          }
        }
      } else {
        var list = [];
        var lastRow = sheet.getLastRow();
        var lastCol = sheet.getLastColumn();
        if (lastRow > 1) {
          var headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
          var values = sheet.getRange(2, 1, lastRow - 1, lastCol).getValues();
          for (var i = 0; i < values.length; i++) {
            var row = values[i];
            var item = {};
            for (var j = 0; j < headers.length; j++) {
              var colName = headers[j];
              var cellVal = row[j];
              
              var fieldName = colName;
              if (fieldName === "ID") fieldName = "id";
              else if (fieldName === "Email") fieldName = "email";
              
              if (typeof cellVal === "string" && (cellVal.indexOf("[") === 0 || cellVal.indexOf("{") === 0)) {
                try {
                  item[fieldName] = JSON.parse(cellVal);
                } catch(err) {
                  item[fieldName] = cellVal;
                }
              } else {
                item[fieldName] = cellVal;
              }
            }
            list.push(item);
          }
        }
        result[name] = list;
      }
    }
    return ContentService.createTextOutput(JSON.stringify({result: "success", data: result})).setMimeType(ContentService.MimeType.JSON);
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({result: "error", message: error.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e) {
  try {
    var json = JSON.parse(e.postData.contents);
    var type = json.type;
    var action = json.action || "upsert";
    var data = json.data;
    
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var isConfig = ["settings", "seo", "robots"].indexOf(type) !== -1;
    
    if (isConfig) {
      var sheet = ss.getSheetByName("system_config");
      if (!sheet) {
        sheet = ss.insertSheet("system_config");
        sheet.appendRow(["Key", "Value"]);
      }
      
      var lastRow = sheet.getLastRow();
      var foundIndex = -1;
      if (lastRow > 1) {
        var values = sheet.getRange(1, 1, lastRow, 1).getValues();
        for (var i = 1; i < values.length; i++) {
          if (String(values[i][0]) === type) {
            foundIndex = i + 1;
            break;
          }
        }
      }
      
      var stringVal = typeof data === "string" ? data : JSON.stringify(data);
      if (foundIndex !== -1) {
        sheet.getRange(foundIndex, 2).setValue(stringVal);
      } else {
        sheet.appendRow([type, stringVal]);
      }
    } else {
      var sheet = ss.getSheetByName(type);
      var uniqueKey = type === "subscribers" ? "email" : "id";
      var idValue = uniqueKey === "id" ? String(data.id || "") : String(data.email || "").toLowerCase();
      
      var headers = [];
      if (!sheet) {
        sheet = ss.insertSheet(type);
        headers.push(uniqueKey === "id" ? "ID" : "Email");
        for (var k in data) {
          if (k !== uniqueKey) headers.push(k);
        }
        sheet.appendRow(headers);
      } else {
        var lastCol = sheet.getLastColumn();
        if (lastCol > 0) {
          headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0];
        } else {
          headers.push(uniqueKey === "id" ? "ID" : "Email");
          sheet.appendRow(headers);
        }
      }
      
      var lastCol = sheet.getLastColumn();
      var headersModified = false;
      for (var k in data) {
        var headerIndex = headers.indexOf(k);
        if (k === "id" && headers.indexOf("ID") !== -1) continue;
        if (k === "email" && headers.indexOf("Email") !== -1) continue;
        
        if (headerIndex === -1) {
          sheet.getRange(1, lastCol + 1).setValue(k);
          headers.push(k);
          lastCol++;
          headersModified = true;
        }
      }
      
      var rowData = [];
      for (var i = 0; i < headers.length; i++) {
        var h = headers[i];
        if (h === "ID") {
          rowData.push(data.id);
        } else if (h === "Email") {
          rowData.push(data.email);
        } else {
          var val = data[h];
          if (val === null || val === undefined) {
            rowData.push("");
          } else if (typeof val === "object") {
            rowData.push(JSON.stringify(val));
          } else {
            rowData.push(val);
          }
        }
      }
      
      var lastRow = sheet.getLastRow();
      var foundIndex = -1;
      if (lastRow > 1) {
        var values = sheet.getRange(1, 1, lastRow, 1).getValues();
        for (var i = 1; i < values.length; i++) {
          if (String(values[i][0]).toLowerCase() === idValue.toLowerCase()) {
            foundIndex = i + 1;
            break;
          }
        }
      }
      
      if (action === "delete") {
        if (foundIndex !== -1) {
          sheet.deleteRow(foundIndex);
        }
      } else {
        if (foundIndex !== -1) {
          sheet.getRange(foundIndex, 1, 1, rowData.length).setValues([rowData]);
        } else {
          sheet.appendRow(rowData);
        }
      }
    }
    
    return ContentService.createTextOutput(JSON.stringify({result: "success"})).setMimeType(ContentService.MimeType.JSON);
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({result: "error", message: error.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}

Supabase Integration Setup

      
    

    

      
Follow these steps to synchronize your website database in real-time using Supabase:

      
    
        
  1. Create a free account on Supabase.
    2. 
        
  2. Create a new project (e.g. Travel Platform).
    3. 
        
  3. Once the project is ready, go to the SQL Editor in the left sidebar and click New Query.
    4. 
        
  4. Copy and paste the following SQL script to create your tables:
          
    -- Config table
CREATE TABLE IF NOT EXISTS system_config (key text PRIMARY KEY, value jsonb);

-- List tables (id + data wrapper)
CREATE TABLE IF NOT EXISTS tours        (id bigint PRIMARY KEY, data jsonb);
